break

这 break语句用于过早结束 for，repeat ，while或者 do / until某种循环，或告诉一个 switch声明到此结束，或过早结束 with称呼。请参阅各个页面以了解这些不同的功能，以更深入地说明在每种情况下如何使用它。请注意，如果 break在任何这些上下文之外使用它会给出错误。

在下面，您可以看到一些有关如何 break可以使用，其语法很简单：

break;

break在一个 for环形：

for (var i = 0; i < 10; i += 1)
    {
    if (array[i] == 234)
        {
        break;
        }
    }
num = i;

break在一个 repeat环形：

var i = 0;
var temp = 0;
repeat (10)
    {
    temp += array[i];
    if (temp > max_total)
        {
        break;
        }
    else
        {
        i += 1;
        }
    }

break在一个 while环形：

var i = 0;
while (!place_free(x, y))
    {
    x = random(room_width);
    y = random(room_height);
    if (i > 50)
        {
        break;
        }
    else
        {
        i+=1;
        }
    }

break在一个 do / until环形：

var _id = noone;
do
    {
    _id = list[| 0];
    if instance_exists(_id)
        {
        _break;
        }
    ds_list_delete(list, 0);
    }
until (ds_list_empty(list));
target = _id;

break使用时 with：

var count = 0;
with (obj_Enemy)
    {
    count++;
    if (count > 10)
        {
        break;
        }
    hp = 100;
    }

break在一个 switch：

switch (keyboard_key)
    {
    case vk_left:
    case ord("A"):
        x -= 4;
        break;
    case vk_right:
    case ord("D"):
        x += 4;
        break;
    case vk_up:
    case ord("W"):
        y -= 4;
        break;
    case vk_down:
    case ord("S"):
        y += 4;
        break;
    }
